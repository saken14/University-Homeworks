#include <iostream>
#include <cmath>
#include <math.h>
#include <string>
#include <windows.h>
#include <stdio.h>
using namespace std;

int main () {
    setlocale(LC_ALL, "RUS");
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
    string s;
    getline(cin, s);
    char c;
    int len = s.length(), counter=0,x=0,n=0, q=0,p=0, min, u;

    for (int i=0; i<len; i++){
        if (s[i] == ' ')
            counter++;
    }
    counter++;
    int a[counter];

    for (int i=0; i<len; i++){
        if (s[i] == ' '){
            x=i-p;
            a[q]=x;
            q++;
            p=i+1;
        }
}
    u = s.rfind(' ');
    a[counter-1]=len-u-1;
    min=a[0];
    for (int i=0; i<sizeof(a)/4; i++){
        if (min>a[i])

    getline(cin, s);
    int len = s.length(), n=0, x=0, y=0;

    for(int i=0; i<len; i++){
        if(s[i]=='('){
            s1[n]=s[i];
            n++;
        }
    }
    n=0;
    for(int i=0; i<len; i++){
        if(s[i]==')'){
            s2[n]=s[i];
            n++;
        }
    }

        if(s1[i]=='('){
            x++;
           }

           }

return 0;
