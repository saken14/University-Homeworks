//////////////////////////////////// Торговые сделки в Амбере/////////////////////////////////////////
#include <iostream>
using namespace std;
int number_system(int n, int p) {
	int arr[100] = {}, m, c = 0, res = 0;
	m = n;
	for (int i = 0; i < n; i++) {
		arr[i] = m % p;
		m /= p;
		c++;
		if (m == 0)
			break;
	}
	for (int i = 0; i < c; i++) {
		res += arr[i];
	}

	return res;
}

int main() {
	int n1, p1, n2, p2;
	cin >> n1 >> p1 >> n2 >> p2;

	int res1 = number_system(n1,p1);
	int res2 = number_system(n2, p2);

	if (res1 == res2)
		cout << res1;
	else
		cout << 0;

	return 0;
}

//////////////////////////////////// Азартный Шрэк/////////////////////////////////////////

#include <iostream>
using namespace std;

int main() {
	int n, arr[100] = {}, minr = 0, maxr = 0;
	cin >> n;

	for (int i = 0; i < n; i++)
		cin >> arr[i];
	
	for (int i = 0; i < n-1; i++)//3 1 8 100	
		for (int j = 0; j < n-1; j++)
			if (arr[j] > arr[j + 1])
				swap(arr[j], arr[j + 1]);

	for (int i = 0; i < n; i++) {
		if (i < n / 2)
			minr += arr[i];
		else
			maxr += arr[i];
	}
	cout << maxr - minr;
	return 0;
}

//////////////////////////////////// Выборы/////////////////////////////////////////

#include <iostream>
using namespace std;

int main() {
	int n, arr[1001] = {}, r = 0;
	cin >> n;

	for (int i = 0; i < n; i++)
		cin >> arr[i];

	for (int i = 0; i < n; i++)
		for (int j = 0; j < n-1; j++) // 4 2 1 3 7
			if (arr[j] > arr[j + 1])
				swap(arr[j], arr[j + 1]);
	
	for (int i = 0; i < n / 2 + 1; i++) {
		if (arr[i] == 1 || arr[i] == 2)
			r += arr[i];
		else
			r += arr[i] / 2 + 1;
	}
	cout << r;
	return 0;
}

//////////////////////////////////// Миша и негатив/////////////////////////////////////////

#include <iostream>
using namespace std;

int main() {
	int n, m, c = 0;
	char arr1[100][100] = {}, arr2[100][100] = {};
	cin >> n >> m;

	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> arr1[i][j];
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> arr2[i][j];
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			if (arr1[i][j] != arr2[i][j])
				c++;
	cout << n*m - c;

	return 0;
}
